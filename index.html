/** * Q-SAFE Academy - Backend Engine * Desarrollado para Google Apps Script */ const SPREADSHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId(); function doGet() { setupDatabase(); // Asegura que las hojas existan al cargar return HtmlService.createHtmlOutputFromFile('Index') .setTitle('Q-SAFE Academy | Formación Cuántica') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL) .addMetaTag('viewport', 'width=device-width, initial-scale=1'); } /** * Configuración inicial de la base de datos en Google Sheets */ function setupDatabase() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheets = { "Usuarios": ["ID", "Nombre", "Correo", "Plan", "Fecha"], "Pagos": ["ID Pago", "Usuario", "Plan", "Valor", "Método Pago", "Fecha"], "Test_Perfil": ["Usuario", "Objetivo", "Nivel", "Disponibilidad", "Resultado"], "Ranking": ["Usuario", "Puntos", "Posición", "Badge"], "Certificados": ["Usuario", "Curso", "Nivel", "Fecha", "Estado"], "Soporte": ["Usuario", "Mensaje", "Estado", "Fecha"], "Vacantes": ["Empresa", "Cargo", "Descripción", "Link"] }; for (let name in sheets) { let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(sheets[name]); sheet.getRange(1, 1, 1, sheets[name].length).setFontWeight("bold").setBackground("#1e293b").setFontColor("#ffffff"); } } } /** * Función genérica para guardar datos en cualquier hoja */ function saveData(sheetName, dataArray) { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(sheetName); sheet.appendRow(dataArray); return { success: true, message: "Registro exitoso" }; } catch (e) { return { success: false, message: e.toString() }; } } // --- FUNCIONES ESPECÍFICAS --- function registrarUsuario(nombre, correo, plan) { const id = "USR-" + new Date().getTime(); const fecha = new Date().toLocaleString(); return saveData("Usuarios", [id, nombre, correo, plan, fecha]); } function registrarPago(usuario, plan, valor, metodo) { const idPago = "PAY-" + Math.random().toString(36).substr(2, 9).toUpperCase(); const fecha = new Date().toLocaleString(); return saveData("Pagos", [idPago, usuario, plan, valor, metodo, fecha]); } function guardarTest(usuario, objetivo, nivel, disponibilidad, resultado) { return saveData("Test_Perfil", [usuario, objetivo, nivel, disponibilidad, resultado]); } function enviarSoporte(usuario, mensaje) { const fecha = new Date().toLocaleString(); return saveData("Soporte", [usuario, mensaje, "Abierto", fecha]); } /** * Obtener datos para el Dashboard */ function getDashboardData() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const ranking = ss.getSheetByName("Ranking").getDataRange().getValues().slice(1); const vacantes = ss.getSheetByName("Vacantes").getDataRange().getValues().slice(1); return { ranking: ranking.map(r => ({ usuario: r[0], puntos: r[1], pos: r[2], badge: r[3] })), vacantes: vacantes.map(v => ({ empresa: v[0], cargo: v[1], desc: v[2], link: v[3] })) }; }
